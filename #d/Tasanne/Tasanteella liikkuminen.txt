    void OnTriggerStay(Collider other)
        {

            if (other.gameObject.tag == "platform")
            {
                Debug.Log("Nyt Vaihtuu Controlli");
                transform.parent = other.transform;

            }
        }

        void OnTriggerExit(Collider other)
        {
            if (other.gameObject.tag == "platform")
            {
                transform.parent = null;

            }
        }
